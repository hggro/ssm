数据库
  CREATE TABLE user (
id  INT NOT NULL AUTO_INCREMENT COMMENT '编号',
username varchar(50) NOT NULL COMMENT '登录名',
password INT NOT NULL COMMENT '密码',
usertype varchar(50) NOT NULL COMMENT '用户类型',
KEY id (id)
) ENGINE=INNODB DEFAULT CHARSET=utf8
